Q1 (User Inputs, Operations & Output)
Write a program to input two numbers and display their sum.*/

#include <stdio.h>

int main() {
    int a, b, s;

    printf("Enter first number: ");
    scanf("%d", &a);

    printf("Enter second number: ");
    scanf("%d", &b);

    s = a + b;

    printf("Sum: %d\n", s);

    return 0;
}
