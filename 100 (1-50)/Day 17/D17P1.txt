/*
Q33: Write a program to check if a number is an Armstrong number.

Sample Test Cases:
Input 1:
153
Output 1:
Armstrong

Input 2:
123
Output 2:
Not Armstrong
*/

#include <stdio.h>
#include <math.h>

int main() {
    int num, temp, remainder, n = 0;
    double sum = 0.0;

    printf("Enter a number: ");
    scanf("%d", &num);

    temp = num;

    // Count number of digits
    while (temp != 0) {
        n++;
        temp /= 10;
    }

    temp = num;

    // Compute sum of digits raised to power n
    while (temp != 0) {
        remainder = temp % 10;
        sum += pow(remainder, n);
        temp /= 10;
    }

    if ((int)sum == num)
        printf("Armstrong\n");
    else
        printf("Not Armstrong\n");

    return 0;
}
