/*
Q41: Write a program to swap the first and last digit of a number.

Sample Test Cases:
Input 1:
1234
Output 1:
4231

Input 2:
1001
Output 2:
1001
*/

#include <stdio.h>
#include <math.h>

int main() {
    int num, first, last, digits, result;

    printf("Enter a number: ");
    scanf("%d", &num);

    last = num % 10;
    digits = (int)log10(num); // number of digits - 1
    first = num / pow(10, digits);

    // Remove first and last digits
    result = num % (int)pow(10, digits); // remove first digit
    result = result / 10; // remove last digit

    // Construct new number with swapped first and last
    result = last * pow(10, digits) + result * 10 + first;

    printf("%d\n", result);

    return 0;
}
