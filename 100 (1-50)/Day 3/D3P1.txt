/*
Q5: Write a program to convert temperature from Celsius to Fahrenheit.

Sample Test Cases:
Input 1:
0
Output 1:
Fahrenheit=32

Input 2:
100
Output 2:
Fahrenheit=212
*/

#include <stdio.h>

int main() {
    float cel, fahr;
    printf("Hey! enter Celsius temp: ");
    scanf("%f", &cel);
    fahr = (cel * 9 / 5) + 32;
    printf("So, Fahrenheit is roughly %.2f\n", fahr);
    return 0;
}
