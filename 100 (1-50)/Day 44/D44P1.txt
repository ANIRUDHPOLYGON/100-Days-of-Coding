/*
Q87: Count spaces, digits, and special characters in a string.

Sample Test Cases:
Input 1:
a b1&2
Output 1:
Spaces=1, Digits=2, Special=1
*/

#include <stdio.h>

int main() {
    char ggx[300];
    int sp = 0, dg = 0, spc = 0;
    int c;

    int i = 0;
    while ((c = getchar()) != EOF && c != '\n') {
        ggx[i++] = c;
    }
    ggx[i] = '\0';

    for (int k = 0; ggx[k] != '\0'; k++) {
        if (ggx[k] == ' ')
            sp++;
        else if (ggx[k] >= '0' && ggx[k] <= '9')
            dg++;
        else if ((ggx[k] >= 'A' && ggx[k] <= 'Z') || (ggx[k] >= 'a' && ggx[k] <= 'z'))
            ;
        else
            spc++;
    }

    printf("Spaces=%d, Digits=%d, Special=%d", sp, dg, spc);
    return 0;
}
