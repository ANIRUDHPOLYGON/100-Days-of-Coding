/*
Q93: Check if two strings are anagrams of each other.

Sample Test Cases:
Input 1:
listen
silent
Output 1:
Anagrams

Input 2:
hello
world
Output 2:
Not anagrams
*/

#include <stdio.h>
#include <string.h>

int main() {
    char p1z[300], q2r[300];
    scanf("%s", p1z);
    scanf("%s", q2r);

    if (strlen(p1z) != strlen(q2r)) {
        printf("Not anagrams");
        return 0;
    }

    int f1[256] = {0}, f2[256] = {0};

    for (int i = 0; p1z[i] != '\0'; i++) {
        f1[(unsigned char)p1z[i]]++;
        f2[(unsigned char)q2r[i]]++;
    }

    for (int j = 0; j < 256; j++) {
        if (f1[j] != f2[j]) {
            printf("Not anagrams");
            return 0;
        }
    }

    printf("Anagrams");
    return 0;
}
