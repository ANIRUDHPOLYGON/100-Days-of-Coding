/*
Q97: Print the initials of a name.

Sample Test Cases:
Input 1:
John Doe
Output 1:
J.D.
*/

#include <stdio.h>

int main() {
    char nm2[300];
    int c, i = 0;

    while ((c = getchar()) != EOF && c != '\n') {
        nm2[i++] = c;
    }
    nm2[i] = '\0';

    if (nm2[0] != ' ')
        printf("%c.", nm2[0]);

    for (int j = 1; nm2[j] != '\0'; j++) {
        if (nm2[j] == ' ' && nm2[j+1] != '\0' && nm2[j+1] != ' ')
            printf("%c.", nm2[j+1]);
    }

    return 0;
}
