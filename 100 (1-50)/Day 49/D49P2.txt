/*
Q98: Print initials of a name with the surname displayed in full.

Sample Test Cases:
Input 1:
John David Doe
Output 1:
J.D. Doe
*/

#include <stdio.h>
#include <string.h>

int main() {
    char nm9[400], last[200];
    int c, i = 0;

    while ((c = getchar()) != EOF && c != '\n') {
        nm9[i++] = c;
    }
    nm9[i] = '\0';

    int len = strlen(nm9);
    int end = len - 1;

    while (end >= 0 && nm9[end] == ' ') end--;

    int start = end;
    while (start >= 0 && nm9[start] != ' ') start--;

    int p = 0;
    for (int k = start + 1; k <= end; k++)
        last[p++] = nm9[k];
    last[p] = '\0';

    if (nm9[0] != ' ')
        printf("%c.", nm9[0]);

    for (int z = 1; z < start; z++) {
        if (nm9[z] == ' ' && nm9[z+1] != ' ' && nm9[z+1] != '\0')
            printf("%c.", nm9[z+1]);
    }

    printf(" %s", last);

    return 0;
}
