/*
Q99: Change the date format from dd/04/yyyy to dd-Apr-yyyy.

Sample Test Cases:
Input 1:
15/04/2025
Output 1:
15-Apr-2025
*/

#include <stdio.h>

int main() {
    char dt7[20], d1[3], y1[5];
    scanf("%s", dt7);

    d1[0] = dt7[0];
    d1[1] = dt7[1];
    d1[2] = '\0';

    y1[0] = dt7[6];
    y1[1] = dt7[7];
    y1[2] = dt7[8];
    y1[3] = dt7[9];
    y1[4] = '\0';

    printf("%s-Apr-%s", d1, y1);

    return 0;
}
