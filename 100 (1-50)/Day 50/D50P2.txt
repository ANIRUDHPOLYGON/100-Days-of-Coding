/*
Q100: Print all sub-strings of a string.

Sample Test Cases:
Input 1:
abc
Output 1:
a,ab,abc,b,bc,c
*/

#include <stdio.h>
#include <string.h>

int main() {
    char sx1[300];
    scanf("%s", sx1);
    int len = strlen(sx1);

    for (int i = 0; i < len; i++) {
        for (int j = i; j < len; j++) {
            for (int k = i; k <= j; k++)
                printf("%c", sx1[k]);
            if (!(i == len-1 && j == len-1))
                printf(",");
        }
    }

    return 0;
}
