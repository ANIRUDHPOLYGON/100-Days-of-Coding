/*
Q115: Write a program to take two strings s and t as inputs (assume all characters are lowercase). The task is to determine if s and t are valid anagrams, meaning they contain the same characters with the same frequencies. Print "Anagram" if they are, otherwise "Not Anagram".

Sample Test Cases:
Input 1:
s = "anagram", t = "nagaram"
Output 1:
Anagram

Input 2:
s = "rat", t = "car"
Output 2:
Not Anagram
*/

#include <stdio.h>
#include <string.h>

int main() {
    char s[500], t[500];
    scanf("%s %s", s, t);

    int freqS[26] = {0}, freqT[26] = {0};
    for(int i = 0; s[i] != '\0'; i++)
        freqS[s[i]-'a']++;
    for(int i = 0; t[i] != '\0'; i++)
        freqT[t[i]-'a']++;

    int isAnagram = 1;
    for(int i = 0; i < 26; i++) {
        if(freqS[i] != freqT[i]) {
            isAnagram = 0;
            break;
        }
    }

    if(isAnagram)
        printf("Anagram");
    else
        printf("Not Anagram");

    return 0;
}
